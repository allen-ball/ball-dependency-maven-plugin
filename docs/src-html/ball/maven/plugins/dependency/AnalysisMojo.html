050import org.apache.maven.plugin.MojoFailureException;

054import static java.util.stream.Collectors.toMap;
055import static java.util.stream.Collectors.toSet;
056import static org.apache.commons.lang3.StringUtils.EMPTY;
057import static org.apache.maven.plugins.annotations.LifecyclePhase.GENERATE_RESOURCES;
058import static org.apache.maven.plugins.annotations.ResolutionScope.TEST;

060/**
061 * {@link org.apache.maven.plugin.Mojo} to provide analysis of dependencies.
062 *
063 * {@injected.fields}
064 *
065 * {@maven.plugin.fields}
066 *
067 * {@bean.info}
068 *
069 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
070 */
071@Mojo(name = "analysis", requiresDependencyResolution = TEST,
072      defaultPhase = GENERATE_RESOURCES, requiresProject = true)
073@NoArgsConstructor @ToString @Slf4j
074public class AnalysisMojo extends AbstractDependencyMojo {
075    private static Comparator<? super Artifact> COMPARATOR = Comparator.comparing(ArtifactUtils::key);
076
077    /* @CompileTimeCheck */
078    private static final Pattern PATTERN =
079        Pattern.compile("(?i)^(META-INF/versions/[\\p{Digit}]+/)?(?<path>.*)[.]class$");
080
081    @Inject private MavenProject project = null;
082
083    @Override
084    public void execute() throws MojoExecutionException, MojoFailureException {
085        Set<String> scope = getScope();
086        Map<Artifact,Set<String>> artifactClassNamesMap =
087            project.getArtifacts().stream()
088            .filter(t -> scope.contains(t.getScope()))
089            .collect(toMap(k -> k, v -> classesIn(v), (t, u) -> u, () -> new TreeMap<>(COMPARATOR)));
090        /*
091         * Map overlapping classes (by name) to common Set of Artifacts.
092         */
093        Map<Set<String>,Set<Artifact>> map = new HashMap<>();
094        List<Artifact> list = new LinkedList<>(artifactClassNamesMap.keySet());
095
096        while (! list.isEmpty()) {
097            Artifact artifact0 = list.remove(0);
098            Set<String> set0 = artifactClassNamesMap.get(artifact0);
099
100            for (Artifact artifactN : list) {
101                Set<String> setN = artifactClassNamesMap.get(artifactN);
102
103                log.debug(EMPTY);
104
105                if (! Collections.disjoint(set0, setN)) {
106                    Set<String> key = new TreeSet<>();
107
108                    key.addAll(set0);
109                    key.retainAll(setN);
110
111                    Collections.addAll(map.computeIfAbsent(key, k -> new TreeSet<>(COMPARATOR)),
112                                       artifact0, artifactN);
113
114                    if (set0.equals(setN)) {
115                        log.debug("{} and {} contain the same class entries", artifact0, artifactN);
116                    } else if (set0.contains(setN)) {
117                        log.debug("{} fully contains {} class entries", artifact0, artifactN);
118                    } else if (setN.contains(set0)) {
119                        log.debug("{} fully contains {} class entries", artifactN, artifact0);
120                    } else {
121                        log.debug("{} and {} contain {} common class entries", artifact0, artifactN, key.size());
122                    }
123                } else {
124                    log.debug("{} and {} contain disjoint class entries", artifact0, artifactN);
125                }
126            }
127        }
128
129        if (! map.isEmpty()) {
130            for (Map.Entry<Set<String>,Set<Artifact>> entry : map.entrySet()) {
131                Set<String> names = entry.getKey();
132                Set<Artifact> artifacts = entry.getValue();
133
134                log.info(EMPTY);
135
136                int limit = 10;
137
138                names.stream()
139                    .limit(limit)
140                    .forEach(t -> log.info("{}", t));
141
142                if (names.size() > limit) {
143                    log.info("... with {} more classes", names.size() - limit);
144                }
145
146                artifacts.stream()
147                    .forEach(t -> log.info("{} {}", t, t.getDependencyTrail()));
148            }
149        } else {
150            log.info("No artifacts with overlapping classes were detected.");
151        }
152    }
153
154    private List<String> IGNORE = Arrays.asList("module-info");
155
156    private Set<String> classesIn(Artifact artifact) {
157        URL url = toURL(artifact);
158        Set<String> classes = Collections.emptySet();
159
160        try {
161            JarFile jar = ((JarURLConnection) url.openConnection()).getJarFile();
162
163            classes =
164                jar.stream()
165                .map(JarEntry::getName)
166                .map(PATTERN::matcher)
167                .filter(t -> t.matches())
168                .map(t -> t.group("path"))
169                .filter(t -> (! IGNORE.contains(t)))
170                .map(t -> t.replaceAll("/", "."))
171                .collect(toSet());
172        } catch (ZipException exception) {
173        } catch (IOException exception) {
174            log.debug("{}: {}", artifact, exception.getMessage(), exception);
175        }
176
177        return classes;
178    }
179
180    private URL toURL(Artifact artifact) {
181        URL url = null;
182
183        try {
184            url = new URI("jar", artifact.getFile().toURI().toASCIIString() + "!/", null).toURL();
185        } catch(URISyntaxException | MalformedURLException exception) {
186            log.debug("{}: {}", artifact, exception.getMessage(), exception);
187            throw new IllegalStateException(exception);
188        }
189
190        return url;
191    }
192}
